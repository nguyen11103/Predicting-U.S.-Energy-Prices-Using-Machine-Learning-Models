def create_sequences_encoder_decoder(data, n_steps_in=30, n_steps_out=1):
    """
    To sequences cho Encoder-Decoder
    n_steps_in: s bc u vo cho encoder
    n_steps_out: s bc u ra t decoder
    """
    X, y = [], []
    for i in range(n_steps_in, len(data) - n_steps_out + 1):
        # Encoder input: n_steps_in timesteps
        X.append(data[i - n_steps_in:i])
        # Decoder target: n_steps_out timesteps
        y.append(data[i:i + n_steps_out])
    return np.array(X), np.array(y)

class PositionalEncoding(nn.Module):
    def __init__(self, d_model, max_len=5000):
        super(PositionalEncoding, self).__init__()
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model))
        pe[:, 0::2] = torch.sin(position * div_term)
        if d_model > 1:
            pe[:, 1::2] = torch.cos(position * div_term[:pe[:, 1::2].shape[1]])
        pe = pe.unsqueeze(0).transpose(0, 1)
        self.register_buffer('pe', pe)

    def forward(self, x):
        return x + self.pe[:x.size(0), :].transpose(0, 1)

class Transformer(nn.Module):
    def __init__(self, input_dim=1, d_model=64, nhead=2, num_encoder_layers=2,
                 num_decoder_layers=2, dropout=0.1):
        super(Transformer, self).__init__()
        self.d_model = d_model

        # Input embedding
        self.input_embedding = nn.Linear(input_dim, d_model)
        self.target_embedding = nn.Linear(input_dim, d_model)

        # Transformer
        self.transformer = nn.Transformer(
            d_model=d_model,
            nhead=nhead,
            num_encoder_layers=num_encoder_layers,
            num_decoder_layers=num_decoder_layers,
            dropout=dropout,
            batch_first=True
        )

        # Output projection
        self.output_projection = nn.Linear(d_model, input_dim)

        # Initialize weights
        self._init_weights()

    def _init_weights(self):
        for p in self.parameters():
            if p.dim() > 1:
                nn.init.xavier_uniform_(p)

    def forward(self, src, tgt):
        # Embedding
        src_emb = self.input_embedding(src)
        tgt_emb = self.target_embedding(tgt)

        # Create causal mask for target
        tgt_len = tgt.size(1)
        tgt_mask = torch.triu(torch.full((tgt_len, tgt_len), float('-inf')), diagonal=1)
        tgt_mask = tgt_mask.to(tgt.device)

        # Pass through transformer
        output = self.transformer(src_emb, tgt_emb, tgt_mask=tgt_mask)

        # Project to output dimension
        output = self.output_projection(output)

        return output

def create_decoder_input(y, start_token=0.0):
    """To decoder input bng cch shift target v thm start token"""
    batch_size, seq_len, dim = y.shape
    device = y.device  # Ly device t input tensor
    decoder_input = torch.full((batch_size, seq_len, dim), start_token, device=device)
    decoder_input[:, 1:, :] = y[:, :-1, :]
    return decoder_input

d_models = [64, 128]
n_heads = [2, 4]
n_encoder_layers = [1, 2]
n_decoder_layers = [1, 2]
EPOCHS = 200
early_stop_patience = 20
results = []

print(" Bt u Grid Search cho Transformer Encoder-Decoder...")

for d_model in d_models:
    for nhead in n_heads:
        if d_model % nhead != 0:
            continue
        for num_enc_layers in n_encoder_layers:
            for num_dec_layers in n_decoder_layers:
                config_name = f"d_model={d_model}, nhead={nhead}, enc_layers={num_enc_layers}, dec_layers={num_dec_layers}"
                print(f"\n Hun luyn: {config_name}")
                start = time.time()

                # Khi to m hnh
                model = Transformer(
                    d_model=d_model,
                    nhead=nhead,
                    num_encoder_layers=num_enc_layers,
                    num_decoder_layers=num_dec_layers
                ).to(device)

                optimizer = torch.optim.Adam(model.parameters(), lr=1e-4)
                loss_fn = nn.MSELoss()

                X_tr = X_train.to(device)
                y_tr = y_train.to(device)
                decoder_input_tr = decoder_input_train.to(device)
                X_vl = X_val.to(device)
                y_vl = y_val.to(device)
                decoder_input_vl = decoder_input_val.to(device)

                train_losses = []
                val_losses = []
                min_val_loss = float('inf')
                counter = 0
                best_model_state = None

                for epoch in range(EPOCHS):
                    # Training
                    model.train()
                    optimizer.zero_grad()

                    out_train = model(X_tr, decoder_input_tr)
                    loss = loss_fn(out_train, y_tr)
                    loss.backward()

                    # Gradient clipping
                    torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
                    optimizer.step()

                    # Validation
                    model.eval()
                    with torch.no_grad():
                        out_val = model(X_vl, decoder_input_vl)
                        val_loss = loss_fn(out_val, y_vl)

                    train_losses.append(loss.item())
                    val_losses.append(val_loss.item())

                    # Early stopping
                    if val_loss.item() < min_val_loss:
                        min_val_loss = val_loss.item()
                        best_model_state = model.state_dict().copy()
                        counter = 0
                    else:
                        counter += 1
                        if counter >= early_stop_patience:
                            print(f" Early stopping at epoch {epoch+1}")
                            break

                    if (epoch + 1) % 20 == 0:
                        print(f"Epoch {epoch+1}: Train Loss = {loss.item():.6f}, Val Loss = {val_loss.item():.6f}")

                # Load best model
                if best_model_state:
                    model.load_state_dict(best_model_state)

                # Evaluate on validation set
                model.eval()
                with torch.no_grad():
                    val_pred = model(X_val, decoder_input_val).cpu().numpy()
                    val_true = y_val.cpu().numpy()

                val_pred_inv = scaler.inverse_transform(val_pred.reshape(-1, 1))
                val_true_inv = scaler.inverse_transform(val_true.reshape(-1, 1))
                rmse = sqrt(mean_squared_error(val_true_inv, val_pred_inv))
                duration = time.time() - start

                results.append({
                    'd_model': d_model,
                    'nhead': nhead,
                    'num_encoder_layers': num_enc_layers,
                    'num_decoder_layers': num_dec_layers,
                    'rmse': rmse,
                    'time': duration,
                    'train_losses': train_losses,
                    'val_losses': val_losses,
                    'model_state': best_model_state
                })

                print(f" RMSE: {rmse:.4f}, Thi gian: {duration:.1f}s")

def forecast_future_encoder_decoder(model, last_sequence, n_days, scaler):
    """D bo s dng Encoder-Decoder"""
    model.eval()
    temp_input = list(last_sequence.reshape(-1))
    future_output = []

    for _ in range(n_days):
        # Chun b encoder input
        x_input = np.array(temp_input[-n_steps_in:]).reshape(1, n_steps_in, 1)
        x_tensor = torch.tensor(x_input, dtype=torch.float32).to(device)

        # To decoder input (ch c start token)
        decoder_input = torch.zeros(1, 1, 1).to(device)

        with torch.no_grad():
            yhat = model(x_tensor, decoder_input)
            pred_value = yhat[0, -1, 0].cpu().numpy()

        future_output.append(pred_value)
        temp_input.append(pred_value)

    return scaler.inverse_transform(np.array(future_output).reshape(-1, 1))
