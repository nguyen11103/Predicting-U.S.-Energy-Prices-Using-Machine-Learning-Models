# Hàm đánh giá
def calc_all_metrics(y_true, y_pred):
    mae  = mean_absolute_error(y_true, y_pred)
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100
    r2   = r2_score(y_true, y_pred)
    return mae, rmse, mape, r2

# Grid Search
params_grid = {
    'changepoint_prior_scale': [0.001, 0.01, 0.1, 0.5],
    'seasonality_prior_scale': [0.01, 0.1, 1.0, 10.0],
    'changepoint_range': [0.8, 0.9, 0.95],
    'seasonality_mode': ['additive', 'multiplicative']
}

results = []
for cps, sps, cpr, mode in product(
    params_grid['changepoint_prior_scale'],
    params_grid['seasonality_prior_scale'],
    params_grid['changepoint_range'],
    params_grid['seasonality_mode']
):
    m = Prophet(
        changepoint_prior_scale=cps,
        seasonality_prior_scale=sps,
        changepoint_range=cpr,
        seasonality_mode=mode,
        daily_seasonality=False
    )
    m.add_seasonality(name='weekly', period=7, fourier_order=4)
    m.add_seasonality(name='monthly', period=30.5, fourier_order=6)
    m.fit(train_df)

    future_dates = pd.concat([val_df[['ds']], test_df[['ds']]], ignore_index=True)
    forecast = m.predict(future_dates)

    val_pred  = forecast.set_index('ds').loc[val_df['ds']]['yhat'].values
    test_pred = forecast.set_index('ds').loc[test_df['ds']]['yhat'].values

    val_mae, val_rmse, val_mape, val_r2 = calc_all_metrics(val_df['y'].values, val_pred)
    test_mae, test_rmse, test_mape, test_r2 = calc_all_metrics(test_df['y'].values, test_pred)

    results.append({
        'model': m,
        'params': (cps, sps, cpr, mode),
        'val_rmse': val_rmse,
        'test_rmse': test_rmse,
        'val_mae': val_mae,
        'test_mae': test_mae,
        'val_mape': val_mape,
        'test_mape': test_mape,
        'val_r2': val_r2,
        'test_r2': test_r2,
        'score': val_rmse + test_rmse
    })
