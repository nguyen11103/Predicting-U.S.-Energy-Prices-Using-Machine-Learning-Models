# Xử lý feature "Date" thành số ngày từ ngày đầu tiên
def process_date(df):
    df['Date'] = pd.to_datetime(df['Date'], dayfirst=True)
    df['Date_ordinal'] = (df['Date'] - df['Date'].min()).dt.days
    return df

train_8 = process_date(train_8)
val_1 = process_date(val_1)
test_1 = process_date(test_1)

# Các features sử dụng
features = ['Open', 'High', 'Low', 'Vol', 'Change', 'Date_ordinal']
target = ['Price']

# Chuẩn bị scaler cho input features và target
scaler_x = MinMaxScaler()
scaler_y = MinMaxScaler()

X_train = scaler_x.fit_transform(train_8[features])
y_train = scaler_y.fit_transform(train_8[target])

X_val = scaler_x.transform(val_1[features])
y_val = scaler_y.transform(val_1[target])

X_test = scaler_x.transform(test_1[features])
y_test = scaler_y.transform(test_1[target])

# Chuẩn bị sequence input cho LSTM
n_steps_in = 30  # input sequence length
n_steps_out = 1  # output sequence length (1 step ahead)

def create_sequences(X, y, n_steps_in, n_steps_out=1):
    Xs, ys = [], []
    for i in range(len(X) - n_steps_in - n_steps_out + 1):
        Xs.append(X[i:i+n_steps_in])
        ys.append(y[i+n_steps_in:i+n_steps_in+n_steps_out])
    return np.array(Xs), np.array(ys)

X_train_seq, y_train_seq = create_sequences(X_train, y_train, n_steps_in, n_steps_out)
X_val_seq, y_val_seq = create_sequences(X_val, y_val, n_steps_in, n_steps_out)
X_test_seq, y_test_seq = create_sequences(X_test, y_test, n_steps_in, n_steps_out)

# Dataset và DataLoader
class SequenceDataset(Dataset):
    def __init__(self, X, y):
        self.X = torch.tensor(X, dtype=torch.float32)
        self.y = torch.tensor(y, dtype=torch.float32)
    def __len__(self):
        return len(self.X)
    def __getitem__(self, idx):
        return self.X[idx], self.y[idx]

train_dataset = SequenceDataset(X_train_seq, y_train_seq)
val_dataset = SequenceDataset(X_val_seq, y_val_seq)
test_dataset = SequenceDataset(X_test_seq, y_test_seq)

class LSTMEncoderDecoder(nn.Module):
    def __init__(self, input_dim, hidden_dim, num_layers=1):
        super().__init__()
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers
        self.encoder_lstm = nn.LSTM(input_dim, hidden_dim, num_layers, batch_first=True)
        self.decoder_lstm = nn.LSTM(1, hidden_dim, num_layers, batch_first=True)  # decoder input 1-dim (Price)
        self.fc = nn.Linear(hidden_dim, 1)  # output 1-dim Price

    def forward(self, encoder_inputs, decoder_inputs):
        batch_size = encoder_inputs.size(0)
        _, (hidden, cell) = self.encoder_lstm(encoder_inputs)
        decoder_outputs, _ = self.decoder_lstm(decoder_inputs, (hidden, cell))
        outputs = self.fc(decoder_outputs)  # (batch, seq_len, 1)
        return outputs

def train_model(model, train_loader, val_loader, criterion, optimizer, n_epochs=50, patience=5):
    best_val_loss = np.inf
    best_model_state = None
    epochs_no_improve = 0

    train_losses_per_epoch = []
    val_losses_per_epoch = []

    for epoch in range(n_epochs):
        model.train()
        train_losses = []
        for xb, yb in train_loader:
            xb, yb = xb.to(device), yb.to(device)
            optimizer.zero_grad()
            decoder_input = torch.zeros((xb.size(0), 1, 1), dtype=torch.float32).to(device)
            y_pred = model(xb, decoder_input)
            loss = criterion(y_pred, yb)
            loss.backward()
            optimizer.step()
            train_losses.append(loss.item())

        model.eval()
        val_losses = []
        with torch.no_grad():
            for xb, yb in val_loader:
                xb, yb = xb.to(device), yb.to(device)
                decoder_input = torch.zeros((xb.size(0), 1, 1), dtype=torch.float32).to(device)
                y_pred = model(xb, decoder_input)
                loss = criterion(y_pred, yb)
                val_losses.append(loss.item())
        avg_train_loss = np.mean(train_losses)
        avg_val_loss = np.mean(val_losses)

        train_losses_per_epoch.append(avg_train_loss)
        val_losses_per_epoch.append(avg_val_loss)

        print(f"Epoch {epoch+1}/{n_epochs}, Train Loss: {avg_train_loss:.6f}, Val Loss: {avg_val_loss:.6f}")

        # Early stopping
        if avg_val_loss < best_val_loss:
            best_val_loss = avg_val_loss
            best_model_state = model.state_dict()
            epochs_no_improve = 0
        else:
            epochs_no_improve += 1
            if epochs_no_improve >= patience:
                print("Early stopping")
                break

    model.load_state_dict(best_model_state)
    return model, train_losses_per_epoch, val_losses_per_epoch

    avg_val_loss = np.mean(val_losses_batch)

    if avg_val_loss < best_val_loss:

# Tính các metrics
def calculate_metrics(y_true, y_pred):
    rmse = sqrt(mean_squared_error(y_true, y_pred))
    mae = mean_absolute_error(y_true, y_pred)
    r2 = r2_score(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100
    return rmse, mae, r2, mape

def forecast_future_encoder_decoder(model, last_sequence, n_days, scaler_x, scaler_y):
    model.eval()
    temp_input = last_sequence.copy()
    future_output = []

    for _ in range(n_days):
        x_input = np.array(temp_input[-n_steps_in:]).reshape(1, n_steps_in, len(features))
        x_tensor = torch.tensor(x_input, dtype=torch.float32).to(device)

        decoder_input = torch.zeros(1, 1, 1).to(device)  # start token

        with torch.no_grad():
            yhat = model(x_tensor, decoder_input)
            pred_value = yhat[0, -1, 0].cpu().numpy()

        future_output.append(pred_value)

        new_feat = temp_input[-1].copy()
        new_feat[-1] += 1  # tăng Date_ordinal 1 ngày

        temp_input = np.vstack([temp_input, new_feat])

    forecast_scaled = np.array(future_output).reshape(-1, 1)
    forecast_inversed = scaler_y.inverse_transform(forecast_scaled)
    return forecast_inversed
