def convert_vol(val):
    if isinstance(val, str):
        val = val.replace(',', '')
        if 'K' in val:
            return float(val.replace('K','')) * 1e3
        elif 'M' in val:
            return float(val.replace('M','')) * 1e6
        elif 'B' in val:
            return float(val.replace('B','')) * 1e9
        else:
            return float(val)
    return val

# ===== Hm to sliding window a bin =====
def create_dataset(X, y, time_step=1):
    dataX, dataY = [], []
    for i in range(len(X) - time_step - 1):
        dataX.append(X[i:i+time_step])
        dataY.append(y[i+time_step])
    return np.array(dataX), np.array(dataY)

# ===== M hnh GRU =====
class GRUModel(nn.Module):
    def __init__(self, input_size, hidden_size=64, num_layers=1, dropout=0.2):
        super(GRUModel, self).__init__()
        self.gru = nn.GRU(input_size=input_size,
                          hidden_size=hidden_size,
                          num_layers=num_layers,
                          batch_first=True,
                          dropout=dropout if num_layers > 1 else 0)
        self.fc = nn.Linear(hidden_size, 1)

    def forward(self, x):
        out, _ = self.gru(x)
        out = out[:, -1, :]
        out = self.fc(out)
        return out

# ===== Grid Search =====
hidden_sizes = [32, 64]
num_layers_list = [1, 2]
dropouts = [0.1, 0.3]
lrs = [0.001, 0.0001]
epochs = 50
patience = 10

results = []

for hidden_size, num_layers, dropout, lr in product(hidden_sizes, num_layers_list, dropouts, lrs):
    print(f"ang th: hidden_size={hidden_size}, num_layers={num_layers}, dropout={dropout}, lr={lr}")

    model = GRUModel(input_size=input_size, hidden_size=hidden_size, num_layers=num_layers, dropout=dropout).to(device)
    criterion = nn.MSELoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)

    best_val_loss = float('inf')
    epochs_no_improve = 0

    for epoch in range(epochs):
        model.train()
        optimizer.zero_grad()
        output = model(X_train_tensor)
        loss = criterion(output, y_train_tensor)
        loss.backward()
        optimizer.step()

    model.eval()
    with torch.no_grad():
        val_pred = model(X_val_tensor)
        val_loss = criterion(val_pred, y_val_tensor).item()

    # EarlyStopping
    if val_loss < best_val_loss:
        best_val_loss = val_loss
        epochs_no_improve = 0
    else:
        epochs_no_improve += 1
        if epochs_no_improve >= patience:
            print(f"Early stopping ti epoch {epoch+1} vi val_loss={val_loss:.6f}")
            break

    results.append({
        'hidden_size': hidden_size,
        'num_layers': num_layers,
        'dropout': dropout,
        'lr': lr,
        'val_loss': best_val_loss
    })

# Hm d bo n ngy tip theo
def forecast_future_gru(model, last_sequence, n_days, scaler, n_steps_in):
    model.eval()
    temp_input = list(last_sequence.reshape(-1, last_sequence.shape[-1]))
    future_output = []

    for _ in range(n_days):
        x_input = np.array(temp_input[-n_steps_in:]).reshape(1, n_steps_in, -1)
        x_tensor = torch.tensor(x_input, dtype=torch.float32).to(device)

        with torch.no_grad():
            yhat = model(x_tensor)
            pred_value = yhat.cpu().numpy().flatten()[0]

        future_output.append(pred_value)

        # To bc tip theo vi cc c trng khng i, ch thay i Price (index = 0)
        next_step = temp_input[-1].copy()
        next_step[0] = pred_value
        temp_input.append(next_step)

    return scaler.inverse_transform(np.array(future_output).reshape(-1, 1))
