Cấu hình thực nghiệm: Dự báo giá Dầu thô (Crude Oil) sử dụng SVR¶

Mô hình: Support Vector Regression (SVR)

  • Tập dữ liệu: Giá dầu thô lịch sử (đã xử lý sẵn)

  • Tỉ lệ chia tập: 70% Train – 15% Validation – 15% Test (7:1.5:1.5)

  • Số dòng dữ liệu gốc:

    • Train: 1,618
    • Validation: 346
    • Test: 348

  • Đặc trưng đầu vào:

    • Sử dụng chuỗi thời gian đơn biến Price
    • Không sử dụng đặc trưng kỹ thuật bổ sung

  • Chiến lược dự báo & đánh giá:

    • Chiến lược: Multi-step Forecast cố định (30, 60, 90 ngày tới)
    • Chỉ số đánh giá: MAE, RMSE, MAPE, R²
    • Mục tiêu: Dự báo biến động giá dầu thô theo thời gian

n_train, n_val = len(train), len(val)
train_scaled = scaled_full[:n_train]
val_scaled = scaled_full[n_train:n_train + n_val]
test_scaled = scaled_full[n_train + n_val:]

def create_supervised(data, time_step=30):
    X, y = [], []
    for i in range(len(data) - time_step):
        X.append(data[i:i+time_step, 0])
        y.append(data[i+time_step, 0])
    return np.array(X), np.array(y)

time_step = 30
X_train, y_train = create_supervised(train_scaled, time_step)
X_val, y_val = create_supervised(np.vstack((train_scaled[-time_step:], val_scaled)), time_step)
X_test, _ = create_supervised(np.vstack((val_scaled[-time_step:], test_scaled)), time_step)

val_actual = val['Price'].values[time_step:]
test_actual = test['Price'].values[time_step:]

3. Train các mô hình SVR và chọn theo val¶

In [6]:
# Các mô hình SVR
kernels = {
    'linear': SVR(kernel='linear', C=100),
    'rbf_1': SVR(kernel='rbf', C=100, epsilon=0.0001, gamma=10),
    'rbf_2': SVR(kernel='rbf', C=10000, epsilon=0.0001, gamma=100),
    'rbf_3': SVR(kernel='rbf', C=1000, epsilon=0.0005, gamma=5)
}

results = {}
for name, model in kernels.items():
    model.fit(X_train, y_train)
    val_pred = model.predict(X_val)
    val_pred_inv = scaler.inverse_transform(val_pred.reshape(-1, 1)).flatten()
    val_pred_inv = val_pred_inv[-len(val_actual):]

    val_rmse = np.sqrt(mean_squared_error(val_actual, val_pred_inv))
    val_mae = mean_absolute_error(val_actual, val_pred_inv)
    val_mape = np.mean(np.abs((val_actual - val_pred_inv) / val_actual)) * 100

    results[name] = {
        'model': model,
        'val_pred': val_pred_inv,
        'val_rmse': val_rmse,
        'val_mae': val_mae,
        'val_mape': val_mape
    }

def forecast_future_univariate(model, last_prices, days=90, time_step=30, scaler_y=None):
    history = list(last_prices)
    preds = []
    for _ in range(days):
        x_input = np.array(history[-time_step:]).reshape(1, -1)
        yhat = model.predict(x_input)[0]
        preds.append(yhat)
        history.append(yhat)
    if scaler_y:
        preds = scaler_y.inverse_transform(np.array(preds).reshape(-1, 1)).flatten()
    else:
        preds = np.array(preds)
    start_date = pd.to_datetime(test.index[-1]) + pd.Timedelta(days=1)
    forecast_index = pd.date_range(start=start_date, periods=days)
    return pd.Series(preds, index=forecast_index)
