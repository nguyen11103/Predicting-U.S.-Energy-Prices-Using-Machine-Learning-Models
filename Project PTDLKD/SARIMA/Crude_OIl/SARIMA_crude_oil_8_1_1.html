ts_train = train['Price']
ts_val = val['Price']
ts_all = pd.concat([ts_train, ts_val])

p = d = q = range(0, 3)
P = D = Q = range(0, 2)
m = 7

pdq = list(itertools.product(p, [1], q))
seasonal_pdq = list(itertools.product(P, [1], Q, [m]))

best_aic = float("inf")
best_order = None
best_seasonal_order = None

print(" Tuning SARIMA parameters based on AIC...")

for order in pdq:
    for seasonal_order in seasonal_pdq:
        try:
            model = SARIMAX(ts_train,
                            order=order,
                            seasonal_order=seasonal_order,
                            enforce_stationarity=False,
                            enforce_invertibility=False)
            results = model.fit(disp=False)
            aic = results.aic
            if aic < best_aic:
                best_aic = aic
                best_order = order
                best_seasonal_order = seasonal_order
                print(f" New best AIC: {aic:.2f} - order={order}, seasonal_order={seasonal_order}")
        except:
            continue

print(f"\n Best SARIMA by AIC: order={best_order}, seasonal_order={best_seasonal_order}, AIC={best_aic:.2f}")

def evaluate(true, pred):
    rmse = np.sqrt(mean_squared_error(true, pred))
    mae = mean_absolute_error(true, pred)
    mape = np.mean(np.abs((true - pred) / true)) * 100
    return rmse, mae, mape
