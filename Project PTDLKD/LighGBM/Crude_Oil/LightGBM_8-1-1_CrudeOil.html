def add_features(df):
    df['dayofweek'] = df['Date'].dt.dayofweek
    df['lag1'] = df['Price'].shift(1)
    df['lag2'] = df['Price'].shift(2)
    df['rolling3'] = df['Price'].rolling(3).mean()
    df['rolling7'] = df['Price'].rolling(7).mean()
    df['ema12'] = df['Price'].ewm(span=12).mean()
    df['ema26'] = df['Price'].ewm(span=26).mean()
    df['macd'] = df['ema12'] - df['ema26']
    df['momentum'] = df['Price'] - df['Price'].shift(5)
    return df.dropna()

# Dự đoán 30, 60, 90 ngày
def forecast_lightgbm_multi(model, last_df, n_days, features):
    last_days = last_df.copy()
    forecast = []
    for _ in range(n_days):
        date = last_days.iloc[-1]['Date'] + pd.Timedelta(days=1)

        if len(last_days) < 7:
            raise ValueError("Không đủ dữ liệu để tính rolling hoặc momentum!")

        ema12_series = last_days['Price'].ewm(span=12).mean()
        ema26_series = last_days['Price'].ewm(span=26).mean()

        row = {
            'Date': date,
            'dayofweek': date.dayofweek,
            'lag1': last_days.iloc[-1]['Price'],
            'lag2': last_days.iloc[-2]['Price'],
            'rolling3': last_days['Price'].tail(3).mean(),
            'rolling7': last_days['Price'].tail(7).mean(),
            'ema12': ema12_series.iloc[-1],
            'ema26': ema26_series.iloc[-1],
            'macd': ema12_series.iloc[-1] - ema26_series.iloc[-1],
            'momentum': last_days['Price'].iloc[-1] - last_days['Price'].iloc[-6]
        }

        X_pred = pd.DataFrame([row])[features]
        y_pred = model.predict(X_pred)[0]
        forecast.append({'Date': date, 'yhat': y_pred})

        last_days = pd.concat([last_days, pd.DataFrame([{'Date': date, 'Price': y_pred}])], ignore_index=True)
    return pd.DataFrame(forecast)

# Đánh giá
def calc_metrics_all(y_true, y_pred, name=''):
    mae = mean_absolute_error(y_true, y_pred)
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100
    r2 = r2_score(y_true, y_pred)
    print(f"\n📊 {name}")
    print(f"  MAE : {mae:.4f}")
    print(f"  RMSE: {rmse:.4f}")
    print(f"  MAPE: {mape:.2f}%")
    print(f"  R^2 : {r2:.2%}")

val_pred = model.predict(X_val)
test_pred = model.predict(X_test)

calc_metrics_all(y_val, val_pred, 'Validation Set')
calc_metrics_all(y_test, test_pred, 'Test Set')
