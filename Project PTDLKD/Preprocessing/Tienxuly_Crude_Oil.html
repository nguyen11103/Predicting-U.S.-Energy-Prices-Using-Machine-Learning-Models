    df[col] = df[col].astype(str).str.replace(',', '').astype(float)

# Xử lý cột Vol.
def convert_vol(val):
    if isinstance(val, str):
        val = val.replace(',', '')
        if 'K' in val:
            return float(val.replace('K','')) * 1e3
        elif 'M' in val:
            return float(val.replace('M','')) * 1e6
        elif 'B' in val:
            return float(val.replace('B','')) * 1e9
        else:
            return float(val)
    return val

df['Vol'] = df['Vol'].apply(convert_vol)

# Xử lý cột Change đưa về kiểu float
df['Change'] = df['Change'].astype(str).str.strip().str.replace('%', '', regex=False).astype(float) / 100.0

# Đặt Date làm chỉ mục
df.set_index('Date', inplace=True)

def split_time_series(data, train_ratio, val_ratio, test_ratio):
    n = len(data)
    train_end = int(train_ratio * n)
    val_end = train_end + int(val_ratio * n)
    train = data[:train_end]
    val = data[train_end:val_end]
    test = data[val_end:]
    return train, val, test

# 8-1-1
train_8, val_1, test_1 = split_time_series(df_final, 0.8, 0.1, 0.1)

# 7-1.5-1.5
train_7, val_15, test_15 = split_time_series(df_final, 0.7, 0.15, 0.15)

# 6-2-2
train_6, val_2, test_2 = split_time_series(df_final, 0.6, 0.2, 0.2)
