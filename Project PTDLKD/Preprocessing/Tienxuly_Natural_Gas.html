def convert_vol(val):
    if isinstance(val, str):
        val = val.replace(',', '')
        if 'K' in val:
            return float(val.replace('K','')) * 1e3
        elif 'M' in val:
            return float(val.replace('M','')) * 1e6
        elif 'B' in val:
            return float(val.replace('B','')) * 1e9
        else:
            return float(val)
    return val

df['Vol'] = df['Vol'].apply(convert_vol)

def split_time_series(data, train_ratio, val_ratio, test_ratio):
    n = len(data)
    train_end = int(train_ratio * n)
    val_end = train_end + int(val_ratio * n)
    train = data[:train_end]
    val = data[train_end:val_end]
    test = data[val_end:]
    return train, val, test

# 8-1-1
train_8, val_1, test_1 = split_time_series(df_final, 0.8, 0.1, 0.1)
